    <script>
        let audio = document.getElementById('mainAudio');
        let timerDisplay = document.getElementById('timer-display');
        let overlay = document.getElementById('overlay');
        let gameControl = document.getElementById('game-control');
        let answerBox = document.getElementById('answer-box');
        let answerText = document.getElementById('answer-text');
        let statusMsg = document.getElementById('status-msg');
        
        let countdown;
        let timeLeft = 15;
        let currentButton = null; // Nút hiện tại đang bấm
        let currentAnswer = "";   // Đáp án hiện tại

        function startGame(noteId, btn) {
            // 1. Thiết lập giao diện
            currentButton = btn;
            currentAnswer = btn.getAttribute('data-answer');
            
            overlay.style.display = 'block';
            gameControl.style.display = 'block';
            answerBox.style.display = 'none';
            statusMsg.innerText = "Đang phát nhạc... 🎶";
            btn.classList.add('playing'); // Hiệu ứng rung

            // 2. Phát nhạc
            audio.src = `note${noteId}.mp3`;
            audio.play().catch(error => {
                alert("Lỗi: Không tìm thấy file note" + noteId + ".mp3");
                closeGame();
            });

            // 3. Chạy đồng hồ
            timeLeft = 15;
            timerDisplay.innerText = timeLeft;
            
            clearInterval(countdown); // Xóa đồng hồ cũ nếu có
            countdown = setInterval(() => {
                timeLeft--;
                timerDisplay.innerText = timeLeft;

                if (timeLeft <= 0) {
                    timeOut();
                }
            }, 1000);
        }

        function timeOut() {
            clearInterval(countdown);
            audio.pause(); // Dừng nhạc
            timerDisplay.innerText = "Hết giờ!";
            statusMsg.innerText = "⏰ Đã hết thời gian suy nghĩ!";
            currentButton.classList.remove('playing');
        }

        function revealAnswer() {
            clearInterval(countdown);
            audio.pause(); // Dừng nhạc nếu chưa hết giờ
            audio.currentTime = 0; 
            
            statusMsg.innerText = "🎉 Kết Quả 🎉";
            answerText.innerText = currentAnswer;
            answerBox.style.display = 'block';
            currentButton.classList.remove('playing');
            
            // Đánh dấu nút đã chơi xong (làm mờ)
            currentButton.classList.add('disabled');
            currentButton.onclick = null; // Không cho bấm lại
        }

        function closeGame() {
            clearInterval(countdown);
            audio.pause();
            audio.currentTime = 0;
            
            overlay.style.display = 'none';
            gameControl.style.display = 'none';
            if(currentButton) currentButton.classList.remove('playing');
        }
    </script>
