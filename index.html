<script>
let timerInterval;
let snippetAudio = document.getElementById('snippetAudio');
let fullAudio = document.getElementById('fullAudio');
let bgMusic = document.getElementById('bgMusic');
let currentBtn = null;
let currentNoteId = 0;

bgMusic.volume = 0.2;
snippetAudio.volume = 1.0;
fullAudio.volume = 1.0;

function toggleBgMusic() {
if (bgMusic.paused) bgMusic.play().catch(e => alert("Hãy bấm vào trang web trước!"));
else bgMusic.pause();
}

function playGame(id, btn) {
bgMusic.volume = 0.05;
currentBtn = btn;
currentNoteId = id;
showPopupPhase('question');
document.getElementById('popup-title').innerText = "Lắng nghe nhé... 👂";
document.getElementById('timer').innerText = "15";
snippetAudio.src = `note${id}.mp3`;
snippetAudio.play().catch(e => alert(`Thiếu file note${id}.mp3`));
let timeLeft = 15;
clearInterval(timerInterval);
timerInterval = setInterval(() => {
timeLeft--;
document.getElementById('timer').innerText = timeLeft;
if (timeLeft <= 0) {
clearInterval(timerInterval);
document.getElementById('timer').innerText = "Hết Giờ!";
snippetAudio.pause();
bgMusic.volume = 0.2;
}
}, 1000);
}

function showAnswer() {
clearInterval(timerInterval);
snippetAudio.pause();
let songName = currentBtn.getAttribute('data-song');
document.getElementById('popup-title').innerText = "🎉 KẾT QUẢ 🎉";
document.getElementById('answer-text').innerText = songName;
showPopupPhase('answer');
}

function addFlowerAndPlayFull(teamId) {
let garden = document.getElementById(`garden-${teamId}`);
garden.innerHTML += "🌻 ";
if (currentBtn) {
currentBtn.classList.add('disabled');
currentBtn.onclick = null;
currentBtn.innerHTML = "✅ Xong";
}
playFullSong();
}

function playFullSong() {
bgMusic.pause();
let songName = currentBtn.getAttribute('data-song');
document.getElementById('popup-title').innerText = "🎶 THƯỞNG THỨC 🎶";
document.getElementById('full-song-name').innerText = songName;
showPopupPhase('fullsong');
fullAudio.src = `full${currentNoteId}.mp3`;
fullAudio.play().catch(e => {
alert(`Chưa có file full${currentNoteId}.mp3! Đang quay về màn hình chính.`);
closeAll();
});
fullAudio.onended = function() { closeAll(); };
}

function showPopupPhase(phaseId) {
document.getElementById('modal-overlay').style.display = 'block';
document.getElementById('game-popup').style.display = 'block';
document.getElementById('question-phase').style.display = 'none';
document.getElementById('answer-phase').style.display = 'none';
document.getElementById('fullsong-phase').style.display = 'none';
document.getElementById(phaseId + '-phase').style.display = 'block';
}

function closeAll() {
clearInterval(timerInterval);
snippetAudio.pause();
fullAudio.pause();
bgMusic.volume = 0.2;
if (!bgMusic.paused || bgMusic.currentTime > 0) {
bgMusic.play().catch(()=>{});
}
document.getElementById('modal-overlay').style.display = 'none';
document.getElementById('game-popup').style.display = 'none';
}
